Usage Examples

Dynamic Import for 3D Components

import dynamic from 'next/dynamic';

const TonnetzGrid = dynamic(
    () => import('@/components/mpn-lab/MPNExperiment_TonnetzGrid'),
    { ssr: false, loading: () => <LoadingPlaceholder /> }
);

// In JSX:
<TonnetzGrid trauma={0.7} entropy={0.5} />

Using ConductorScore

import ConductorScoreView from '@/components/mpn-lab/ConductorScore';

<ConductorScoreView
    frame={scoreFrame}
    showChordSymbols={true}
    showDynamics={true}
    isPlaying={isPlaying}
    currentBeat={currentFrameIndex}
/>

Using ScoreOrchestrator

import { getOrchestrator } from '@/components/mpn-lab/score_orchestrator';

const orchestrator = getOrchestrator();
orchestrator.registerActor(actorProfile);
const output = await orchestrator.processFrame(script, trauma, entropy);
